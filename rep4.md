# 4주차 나만의 도형 만들기
## 안동대학교 김민정
```

translate(50,50);  //크기

fill(125,125,0);  //얼굴 색
ellipse(0, 0, 80, 80); //얼굴
fill(0);  //눈 색
ellipse(-20, -15, 25, 25); //눈
ellipse(+20, -15, 25, 25); //눈
fill(225); //눈동자 색
ellipse(-20, -15, 8, 8); //눈동자
ellipse(+20, -15, 8, 8); //눈동자
fill(255,0,0);  //코,입 색
ellipse(0, 0,10,10);  //코
arc(0, 8, 60, 50, 0, PI); //입

```
